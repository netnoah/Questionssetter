# AI Quiz Generator Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Claude Code skill that generates interactive HTML quiz files from configurable question type templates.

**Architecture:** A Claude Code custom command (`/出题`) reads YAML question-type configs, uses AI to generate structured JSON questions, injects them into a self-contained HTML template, and opens the result in the browser. The HTML file includes all CSS/JS for interactive quiz-taking with immediate feedback, collapsible hints, SVG diagram support, and a statistics report.

**Tech Stack:** Claude Code custom commands, YAML configs, vanilla HTML/CSS/JS (zero dependencies)

**Spec:** `docs/superpowers/specs/2026-05-06-quiz-app-design.md`

**Security note:** Question/hint text is rendered as HTML via innerHTML (to support math formatting like `<sub>`, `<sup>`). This is an accepted trust boundary — only the AI generates this content, never end-user input. SVG content is untrusted and is sanitized before rendering.

```yaml
name: 袋鼠数学
description: 袋鼠数学竞赛风格题目
icon: "🦘"

prompt_template: |
  你是一位袋鼠数学竞赛出题专家。请根据以下要求出题：
  主题：{topic}
  数量：{count} 道选择题（每题 5 个选项，A-E）

  出题要求：
  - 题目风格模仿袋鼠数学竞赛：趣味性强、贴近生活、侧重逻辑思维
  - 难度适合对应年级
  - 需要图文配合的题用 SVG 绘制配图（SVG 放在 image 字段）
  - 每题提供详细解析和解题思路（放在 hint 字段）
  - SVG 必须包含 viewBox 属性，只使用基础形状元素（path, rect, circle, ellipse, line, polyline, polygon, text, g）

  按以下格式输出一个 JSON 数组，不要输出任何其他内容：
  {output_format}

output_format:
  description: |
    一个 JSON 数组，每个元素包含：
    - id: 题号（从1开始）
    - question: 题目文本（可包含 HTML 标签用于数学公式）
    - image: SVG 代码字符串（可选，无配图时不要包含此字段）
    - options: 选项数组，每项包含 label（A-E）和 text（选项内容）
    - answer: 正确答案的 label
    - hint: 解析提示（可包含 HTML）
  example: |
    [
      {
        "id": 1,
        "question": "一个长方形的长是12米，宽是8米，面积是多少？",
        "options": [
          {"label": "A", "text": "20平方米"},
          {"label": "B", "text": "40平方米"},
          {"label": "C", "text": "96平方米"},
          {"label": "D", "text": "48平方米"},
          {"label": "E", "text": "84平方米"}
        ],
        "answer": "C",
        "hint": "长方形面积 = 长 × 宽 = 12 × 8 = 96平方米"
      }
    ]

ui:
  primary_color: "#4A90D9"
  option_count: 5
```

Note: `option_style` is removed from this YAML. All question types use vertical option layout for consistency. This simplifies the template and avoids unused config fields.

// ── Sanitize Module ──
const Sanitize = {
  ALLOWED_ELEMENTS: new Set([
    'svg', 'path', 'rect', 'circle', 'ellipse', 'line',
    'polyline', 'polygon', 'text', 'tspan', 'g', 'defs',
    'linearGradient', 'radialGradient', 'stop', 'title', 'desc'
  ]),

  sanitize(svgString) {
    const parser = new DOMParser();
    const doc = parser.parseFromString(svgString, 'image/svg+xml');
    const errorNode = doc.querySelector('parsererror');
    if (errorNode) return '';

    const root = doc.documentElement;
    if (!root.hasAttribute('viewBox')) return '';

    this._cleanNode(root);
    const serializer = new XMLSerializer();
    return serializer.serializeToString(root);
  },

  _cleanNode(node) {
    if (node.nodeType === Node.ELEMENT_NODE) {
      if (!this.ALLOWED_ELEMENTS.has(node.tagName)) {
        node.remove();
        return;
      }
      const attrsToRemove = [];
      for (const attr of node.attributes) {
        if (attr.name.startsWith('on') ||
            ((attr.name === 'href' || attr.name === 'xlink:href') &&
             attr.value.trim().toLowerCase().startsWith('javascript:'))) {
          attrsToRemove.push(attr.name);
        }
      }
      attrsToRemove.forEach(a => node.removeAttribute(a));
    }
    [...node.childNodes].forEach(child => this._cleanNode(child));
  }
};

// ── State Module ──
const State = {
  questions: [],
  meta: {},
  currentIndex: 0,
  answers: {},
  expandedHints: {},
  isReportMode: false,
  isRetryMode: false,
  retryIds: [],

  init() {
    try {
      const metaEl = document.getElementById('quiz-meta');
      const dataEl = document.getElementById('quiz-data');
      this.meta = JSON.parse(metaEl.textContent);
      this.questions = JSON.parse(dataEl.textContent);
    } catch (e) {
      document.getElementById('app').innerHTML =
        '<div class="error-message"><h2>数据加载失败</h2><p>题目数据格式错误，请重新生成。</p></div>';
      throw e;
    }
    this.currentIndex = 0;
    this.answers = {};
    this.expandedHints = {};
    this.isReportMode = false;
    this.isRetryMode = false;
    this.retryIds = [];
  },

  getActiveQuestions() {
    if (this.isRetryMode && this.retryIds.length > 0) {
      return this.questions.filter(q => this.retryIds.includes(q.id));
    }
    return this.questions;
  },

  getCurrentQuestion() {
    const active = this.getActiveQuestions();
    return active[this.currentIndex] || null;
  },

  selectAnswer(questionId, label) {
    if (this.answers[questionId] !== undefined) return;
    this.answers = Object.assign({}, this.answers, { [questionId]: label });
  },

  isAnswered(questionId) {
    return this.answers[questionId] !== undefined;
  },

  isCorrect(questionId) {
    const q = this.questions.find(q => q.id === questionId);
    return q && this.answers[questionId] === q.answer;
  },

  getAnsweredCount() {
    const active = this.getActiveQuestions();
    return active.filter(q => this.isAnswered(q.id)).length;
  },

  allAnswered() {
    return this.getAnsweredCount() === this.getActiveQuestions().length;
  },

  getResults() {
    const active = this.getActiveQuestions();
    const correct = active.filter(q => this.isCorrect(q.id)).length;
    const wrong = active.filter(q => this.isAnswered(q.id) && !this.isCorrect(q.id)).length;
    return {
      total: active.length,
      correct,
      wrong,
      accuracy: active.length > 0 ? Math.round((correct / active.length) * 100) : 0,
      questions: active.map(q => ({
        id: q.id,
        question: q.question,
        correct: this.isCorrect(q.id),
        selected: this.answers[q.id],
        answer: q.answer
      }))
    };
  },

  startRetry() {
    const results = this.getResults();
    const wrongIds = results.questions.filter(r => !r.correct).map(r => r.id);
    if (wrongIds.length === 0) return false;
    this.retryIds = wrongIds;
    // Build new answers object excluding retry question IDs (immutable)
    const retrySet = new Set(wrongIds);
    this.answers = Object.fromEntries(
      Object.entries(this.answers).filter(([id]) => !retrySet.has(Number(id)))
    );
    this.isRetryMode = true;
    this.isReportMode = false;
    this.currentIndex = 0;
    return true;
  }
};
```

// ── Render Module ──
const Render = {
  render() {
    if (State.isReportMode) {
      this.renderReport();
    } else {
      this.renderQuestion();
    }
  },

  renderQuestion() {
    const q = State.getCurrentQuestion();
    if (!q) return;
    const active = State.getActiveQuestions();
    const answered = State.isAnswered(q.id);
    const selected = State.answers[q.id];
    const isCorrect = answered && selected === q.answer;
    const hintExpanded = State.expandedHints[q.id];

    const imageHtml = q.image
      ? '<div class="quiz-image">' + Sanitize.sanitize(q.image) + '</div>'
      : '';

    const optionsHtml = q.options.map(opt => {
      let cls = 'option-btn';
      if (answered) {
        if (opt.label === q.answer) cls += ' correct';
        else if (opt.label === selected) cls += ' wrong';
        cls += ' locked';
      }
      const iconHtml = (answered && opt.label === q.answer)
        ? '<span class="option-icon">✓</span>'
        : (answered && opt.label === selected && opt.label !== q.answer)
          ? '<span class="option-icon">✗</span>'
          : '';
      return '<button class="' + cls + '" data-label="' + opt.label + '"' +
        (answered ? ' disabled' : '') +
        ' aria-label="选项 ' + opt.label + ': ' + opt.text + '">' +
        '<span class="option-label">' + opt.label + '</span>' +
        '<span class="option-text">' + opt.text + '</span>' +
        iconHtml +
        '</button>';
    }).join('');

    const feedbackHtml = answered
      ? '<div class="feedback ' + (isCorrect ? 'correct' : 'wrong') + '">' +

        '<div class="quiz-progress">第 ' + (State.currentIndex + 1) + '/' + active.length + ' 题</div>' +
      '</header>' +
      '<main class="quiz-body">' +
        '<section class="question">' +
          '<p class="question-text">' + (State.currentIndex + 1) + '. ' + q.question + '</p>' +
          imageHtml +
        '</section>' +
        '<fieldset class="options" role="radiogroup">' +
          '<legend class="sr-only">选项</legend>' +
          optionsHtml +
        '</fieldset>' +
        feedbackHtml +
        hintHtml +
      '</main>' +
      '<nav class="quiz-nav" aria-label="题目导航">' +
        navHtml +
        '<div class="nav-buttons">' +
          '<button class="nav-btn" data-action="prev"' + (State.currentIndex === 0 ? ' disabled' : '') + '>← 上一题</button>' +
          '<button class="nav-btn" data-action="next"' + (State.currentIndex === active.length - 1 ? ' disabled' : '') + '>下一题 →</button>' +
        '</div>' +
      '</nav>';
  },

  _renderNav(active) {
    return '<div class="nav-numbers">' + active.map((q, i) => {
      let cls = 'nav-num';
      if (i === State.currentIndex) cls += ' current';
      else if (State.isAnswered(q.id)) {
        cls += State.isCorrect(q.id) ? ' answered-correct' : ' answered-wrong';
      }
      return '<button class="' + cls + '" data-index="' + i + '" aria-label="第' + (i + 1) + '题">' + (i + 1) + '</button>';
    }).join('') + '</div>';
  },

  renderReport() {
    const r = State.getResults();
    const barWidth = r.accuracy;

    const reviewHtml = r.questions.map(item => {
      const icon = item.correct ? '✅' : '❌';
      const detail = item.correct ? '' : ' → ' + item.answer + ' (你选了 ' + item.selected + ')';
      const shortQ = item.question.length > 30 ? item.question.substring(0, 30) + '...' : item.question;
      return '<div class="review-item ' + (item.correct ? 'correct' : 'wrong') + '">' +
        item.id + '. ' + icon + ' ' + shortQ + detail + '</div>';
    }).join('');

    document.getElementById('app').innerHTML =
      '<header class="report-header"><h1>🎉 答题报告</h1></header>' +
      '<main class="report-body">' +
        '<div class="score-section">' +
          '<div class="score-text">正确：' + r.correct + '/' + r.total + '    正确率：' + r.accuracy + '%</div>' +
          '<div class="score-bar"><div class="score-bar-fill" style="width:' + barWidth + '%"></div></div>' +
        '</div>' +
        '<section class="review-list">' + reviewHtml + '</section>' +
        '<div class="action-buttons">' +
          (r.wrong > 0 ? '<button class="action-btn" data-action="retry">重新做错题</button>' : '') +
          '<button class="action-btn" data-action="print">导出为PDF</button>' +
        '</div>' +
      '</main>';
  }
};
```

```javascript
// ── Actions Module ──
const Actions = {
  init() {
    document.getElementById('app').addEventListener('click', (e) => {
      const target = e.target.closest('[data-label], [data-index], [data-action]');
      if (!target) return;

      if (target.dataset.label) {
        this.handleOption(target.dataset.label);
      } else if (target.dataset.index !== undefined) {
        this.handleNav(parseInt(target.dataset.index, 10));
      } else if (target.dataset.action) {
        this.handleAction(target.dataset.action);
      }
    });

    document.addEventListener('keydown', (e) => {
      if (State.isReportMode) return;
      const active = State.getActiveQuestions();
      if (e.key === 'ArrowLeft' || e.key === 'ArrowUp') {
        e.preventDefault();
        if (State.currentIndex > 0) { State.currentIndex--; Render.render(); }
      } else if (e.key === 'ArrowRight' || e.key === 'ArrowDown') {
        e.preventDefault();
        if (State.currentIndex < active.length - 1) { State.currentIndex++; Render.render(); }
      }
    });
  },

  handleOption(label) {
    const q = State.getCurrentQuestion();
    if (!q || State.isAnswered(q.id)) return;
    State.selectAnswer(q.id, label);
    Render.render();
    if (State.allAnswered()) {
      setTimeout(() => {
        State.isReportMode = true;
        Render.render();
      }, 1500);
    }
  },

  handleNav(index) {
    State.currentIndex = index;
    Render.render();
  },

  handleAction(action) {
    switch (action) {
      case 'prev':
        if (State.currentIndex > 0) { State.currentIndex--; Render.render(); }
        break;
      case 'next':
        const active = State.getActiveQuestions();
        if (State.currentIndex < active.length - 1) { State.currentIndex++; Render.render(); }
        break;
      case 'hint': {
        const q = State.getCurrentQuestion();
        if (q) {
          State.expandedHints = Object.assign({}, State.expandedHints, { [q.id]: !State.expandedHints[q.id] });
          Render.render();
        }
        break;
      }
      case 'retry':
        if (State.startRetry()) { Render.render(); }
        break;
      case 'print':
        window.print();
        break;
    }
  }
};

// ── Boot ──
State.init();
Actions.init();
Render.render();
```

Note: The hint toggle button in Render uses `data-action="hint"`, so it's handled by the single delegated listener. No separate event listener needed.
